Binary Tree – City Library Digital Book Index

#include <bits/stdc++.h>
using namespace std;

struct Node {
    string book;
    Node* left;
    Node* right;

    Node(string b) {
        book = b;
        left = right = NULL;
    }
};

// Insert book into BST
Node* insertBook(Node* root, string bookName) {
    if (root == NULL)
        return new Node(bookName);

    if (bookName < root->book)
        root->left = insertBook(root->left, bookName);
    else
        root->right = insertBook(root->right, bookName);

    return root;
}

// Search for a book
bool searchBook(Node* root, string key) {
    if (root == NULL) return false;
    if (root->book == key) return true;

    if (key < root->book)
        return searchBook(root->left, key);
    else
        return searchBook(root->right, key);
}

// Display books alphabetically
void inorder(Node* root) {
    if (root == NULL) return;
    inorder(root->left);
    cout << root->book << endl;
    inorder(root->right);
}

int main() {
    Node* root = NULL;

    // Sample Book List
    vector<string> books = {
        "Data Structures",
        "Operating Systems",
        "Machine Learning",
        "Networking Basics",
        "C++ Programming",
        "Digital Logic"
    };

    // Insert books into the BST
    for (string b : books)
        root = insertBook(root, b);

    cout << "Books in Library (Alphabetical Order):\n";
    inorder(root);

    string searchKey = "Machine Learning";
    cout << "\nSearching for: " << searchKey << endl;

    if (searchBook(root, searchKey))
        cout << "Book FOUND in library index\n";
    else
        cout << "Book NOT FOUND\n";

    return 0;
}
