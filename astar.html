struct Node{
    int x, y, cost, heuristic;
    bool operator<(const Node &other) const {
        return (cost + heuristic) > (other.cost + other.heuristic);
    }
};

int heuristic(int x1,int y1,int x2,int y2){
    return abs(x1 - x2) + abs(y1 - y2); // Manhattan distance
}
