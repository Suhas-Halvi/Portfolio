#include <bits/stdc++.h>
using namespace std;

int main(){
    priority_queue<pair<int,string>> pq;
    pq.push({5,"Zone A"});
    pq.push({2,"Zone B"});
    pq.push({8,"Zone C"});

    while(!pq.empty()){
        auto top = pq.top(); pq.pop();
        cout << top.second << " with priority " << top.first << endl;
    }
}
