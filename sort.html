#include <bits/stdc++.h>
using namespace std;

struct Car {
    string plate;
    int arrival;
};

bool cmp(Car a, Car b){ return a.arrival < b.arrival; }

int main(){
    vector<Car> cars = { {"AB123",10}, {"XY987",5}, {"CD456",7} };
    sort(cars.begin(), cars.end(), cmp);
    for(auto c: cars)
        cout << c.plate << " arrives at " << c.arrival << endl;
}
