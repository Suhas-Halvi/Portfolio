#include <bits/stdc++.h>
using namespace std;

struct Edge {
    int u, v, w;
};

int main() {
    int V = 5, E = 7;
    vector edges = {
        {0,1,6},{0,2,7},{1,2,8},{1,3,5},{1,4,-4},{2,3,-3},{3,1,-2}
    };

    vector dist(V, INT_MAX);
    dist[0]=0;

    for(int i=0;i<V-1;i++){
        for(auto e:edges){
            if(dist[e.u]!=INT_MAX && dist[e.u]+e.w < dist[e.v])
                dist[e.v]=dist[e.u]+e.w;
        }
    }

    for(int i=0;i<V;i++)
        cout << "Node " << i << ": " << dist[i] << endl;
}
