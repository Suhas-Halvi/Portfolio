#include <bits/stdc++.h>
using namespace std;

int main(){
    int pollution[] = {50, 60, 55, 70, 65};
    int n = 5;

    int prefix[n+1];
    prefix[0]=0;
    for(int i=0;i<n;i++)
        prefix[i+1]=prefix[i]+pollution[i];

    int l=1, r=3; // query 1-based
    cout << "Average pollution from day " << l << " to " << r << ": "
         << (prefix[r]-prefix[l-1])/(r-l+1) << endl;
}
