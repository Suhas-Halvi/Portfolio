Smart Tour Routing – Dijkstra Algorithm

#include <bits/stdc++.h>
using namespace std;

// Graph storing adjacency list: attraction -> {adjacent attraction, walking time}
vector<pair<int,int>> graph[100];

void dijkstra(int src, int n) {
    vector<int> dist(n, INT_MAX);
    priority_queue<pair<int,int>, vector<pair<int,int>>, greater<pair<int,int>>> pq;

    dist[src] = 0;
    pq.push({0, src});

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        if (d != dist[u]) continue;

        for (auto &edge : graph[u]) {
            int v = edge.first;
            int w = edge.second;

            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }

    cout << "Fastest walking time from selected starting attraction:\\n";
    for (int i = 0; i < n; i++) {
        cout << "Attraction " << i << " : " << dist[i] << " minutes\\n";
    }
}

int main() {
    int n = 6; 
    // Attractions: 0–Museum, 1–Park, 2–Temple, 3–Fort, 4–Lake, 5–Shopping Street

    graph[0].push_back({1, 10});
    graph[1].push_back({0, 10});

    graph[1].push_back({2, 15});
    graph[2].push_back({1, 15});

    graph[0].push_back({3, 30});
    graph[3].push_back({0, 30});

    graph[2].push_back({3, 5});
    graph[3].push_back({2, 5});

    graph[3].push_back({4, 20});
    graph[4].push_back({3, 20});

    graph[4].push_back({5, 12});
    graph[5].push_back({4, 12});

    graph[2].push_back({5, 25});
    graph[5].push_back({2, 25});

    int start_attraction = 0; // Museum
    dijkstra(start_attraction, n);
}
